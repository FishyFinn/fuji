\LevelThree{stripe}
\LevelFour{Purpose}
Stripe the \ttt{style tags} in \ttt{chat message} sent by players based on permissions.\\

\LevelFour{Example}
\begin{example}{Allow players to use \str{<blue>} tag}
    \ic{/lp group default permission set fuji.style.chat.blue}
\end{example}

\begin{example}{Allow players to use \str{<b>} tag}
    \ic{/lp group default permission set fuji.style.chat.b}
\end{example}

\begin{example}{Allow players to use all tags}
    All tags also including dangerous tags like \str{<click>} tag which can run commands on clicked!\\
    \ic{/lp group default permission set fuji.style.chat.*}
\end{example}
