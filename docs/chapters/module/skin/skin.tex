\LevelOne{skin}
\LevelTwo{Purpose}
This module provides skin customization.

\LevelTwo{Command}
\LevelThree{/skin}

\LevelTwo{Configuration}
\begin{Configuration}
    \item[default\_skins]{
        The default skin used for player who has no skin set.
    }
\end{Configuration}

\LevelTwo{Example}

\begin{example}{Set a skin from mojang}
    \ic{/skin set mojang dream}
\end{example}

\begin{example}{Set a skin from url}
    \ic{/skin set web slim "https://s.namemc.com/i/bd53d152d0cd91d0.png"}
\end{example}

\begin{example}{Clear the skin}
    \ic{/skin clear}
\end{example}
